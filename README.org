** Imports

#+BEGIN_SRC python
from py2neo import Graph
#+END_SRC

** Creds

#+BEGIN_SRC python
host = "127.0.0.1"
port = 7687
user = ""
password = ""
#+END_SRC

** Connect

#+BEGIN_SRC python
graph = Graph(host=host, port=port, user=user, password=password)
#+END_SRC

** Create transaction

#+BEGIN_SRC python
message = "Message"
#+END_SRC

#+BEGIN_SRC python
graph.run("CREATE (a:Greeting) "
          "SET a.message = $message "
          "RETURN a.message + ', from node ' + id(a)", message=message)
#+END_SRC
